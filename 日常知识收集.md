    不能用== ===，因为数组是对象，== ===只能比较两个对象是否具有同一实例
    function equar(a, b) {
        if (a.length !== b.length) {
            return false
        } else {
            for (let i = 0; i < a.length; i++) {
                if (a[i] !== b[i]) {
                    return false
                }
            }
            return true;
        }
    }
